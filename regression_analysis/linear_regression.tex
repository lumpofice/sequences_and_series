    &= \sum(y_{i}^{2}) - (b_{0} + b_{1}\bar{x})\sum(y_{i})\\[2ex]

    SST - SSR &= \sum(y_{i}^{2}) - (b_{0} + b_{1}\bar{x})\sum(y_{i}) - b_{1}^{2}\sum(x_{i} - \bar{x})^{2}\\
    &= \sum(y_{i}^{2}) - b_{0}\sum(y_{i}) - b_{1}\bar{x}\sum(y_{i}) - b_{1}^{2}\sum(x_{i} - \bar{x})^{2}\\

    &= S_{yy} - b_{1}S_{xy} = SSE
\end{align*}
